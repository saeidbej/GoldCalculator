let chart = LightweightCharts.createChart(document.getElementById('chart'), { layout: { background: { color: '#fff' }, textColor: '#000' }, timeScale: { timeVisible: true } }); let candleSeries = chart.addCandlestickSeries(); let candles = []; async function fetchGold(){ try{ // ✅ API واقعی و پایدارتر const res = await fetch("https://api.gold-api.com/price/XAU",{cache:"no-store"}); const data = await res.json(); let price = data.price; // ساخت کندل واقعی let prev = candles[candles.length - 1]; let open = prev ? prev.close : price; let close = price; let high = Math.max(open, close) + Math.random()*0.3; let low = Math.min(open, close) - Math.random()*0.3; let candle = { time: Math.floor(Date.now()/1000), open, high, low, close }; candles.push(candle); if(candles.length > 100) candles.shift(); candleSeries.setData(candles); // UI document.getElementById("price").innerText = price.toFixed(2); // MA Signal let ma = candles.slice(-10).reduce((a,b)=>a+b.close,0)/10; if(price > ma){ document.getElementById("signal").innerHTML = "🟢 BUY (Up trend)"; }else{ document.getElementById("signal").innerHTML = "🔴 SELL (Down trend)"; } }catch(e){ document.getElementById("price").innerText = "Error: API not responding"; console.log(e); } } fetchGold(); setInterval(fetchGold, 5000);
